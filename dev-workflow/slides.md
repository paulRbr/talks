##### `.gitlab-ci.yml`

```yaml
image: ruby:2.4

bundle:
  stage: build
  script:
    - bundle install --deployment
  artifacts:
    paths: [ '.bundle/', 'vendor/' ]

test:
  stage: test
  script:
    - bundle exec rake test

rubocop:
  stage: test
  script:
    - bundle exec rubocop
```
